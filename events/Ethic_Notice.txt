event = {
	id = ethic_notice.3
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		has_global_flag = eci_star
	}
	immediate = {
		every_country = {
			limit = {
				OR ={
					is_gestalt = no
					is_servitor_empire = yes
				}
				OR = {
					NOT = { has_technology = tech_frontier_hospital }
					AND = {
						any_owned_planet = { has_building = building_medical_center }
						NOT = { has_technology = tech_frontier_hospital }
					}
				}
			}
			country_event = { id = ethic_notice.4 }
		}
	}
}

country_event = {
	id = ethic_notice.4
	title = ethic_notice.4.name
	desc = ethic_notice.4.desc
	picture = GFX_evt_ECI
	is_triggered_only = yes
	immediate = {
		if = {
			limit = { NOT = { has_technology = tech_frontier_hospital } }
			give_technology = {
				message = no
				tech = tech_medical_system
			}
		}
		every_owned_planet = {
			limit = {
				has_building = building_medical_center
				planet_owner = {
					NOT = { has_technology = tech_frontier_hospital }
				}
			}
			remove_building = building_medical_center
			add_building = building_clinic
		}
	}
	option = {
		name = ethic_notice.4.a
	}
}
