#朝闻道过劳死

namespace = chao_wen_dao

country_event = {
	id = chao_wen_dao.2
	is_triggered_only = yes
	hide_window = yes

	trigger = {
		OR = {
			has_valid_civic = civic_chaowen_dao
		}
	}

	immediate = {
		else = {
				#改下面这个数字可以改变每次过劳死人数
				while = { count = 2     
					# 献祭人口
					else = {
						random_owned_pop = {
							weights = { base = 2
								modifier = { factor = 4		OR = { is_pop_category = specialist } }	# 专家
								modifier = { factor = 8		OR = { is_pop_category = ruler } }	# 统治者
							}
							kill_pop = yes
						}
					}
				}
				
				set_update_modifiers_batch = end
			}
	}
}
