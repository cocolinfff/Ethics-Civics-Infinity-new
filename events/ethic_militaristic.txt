country_event = {
	id = ethic_militarist.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_policy_flag = first_contact_attack_allowed
		OR = {
			has_ethic = ethic_fanatic_militarist
			is_homicidal = yes
			has_ascension_perk = ap_become_the_crisis
		}
		is_at_war = no
		is_country_type = default
		from = { is_country_type = default }
		NOT = {
			has_country_flag = fervor_event
		}
		if = {
			limit = {
				has_valid_civic = civic_fanatic_purifiers
			}
			from = {
				owner_main_species = {
					NOT = {
						is_same_value = root.owner_main_species
					}
				}
			}
		}
		if = {
			limit = {
				has_valid_civic = civic_machine_terminator
			}
			from = {
				is_machine_empire = no
				is_synthetic_empire = no
			}
		}
	}

	immediate = {
		country_event = {
			id = ethic_militarist.2
			days = 3
			scopes = { from = from }
		}
	}
}

	id = ethic_militaristic_fervor.1
	hide_window = yes
	is_triggered_only = yes

	# This = owner of ship 1 (combatant)
	# From = owner of ship 2 (destroyed)
	# FromFrom = ship 1
	# FromFromFrom = ship 2

	trigger = {
		owner = {
			any_situation = {
				is_situation_type = situation_militaristic_fervor
			}
		}

		# check to exclude starbase kills in the future;
		# right now they're included and the starbase kill objective does nothing,
		# as the menace reward is the same
	}

	immediate = {
		owner = {
			random_situation = {
				limit = {
					is_situation_type = situation_militaristic_fervor
				}
				add_situation_progress = -1
			}
		}
	}
}
