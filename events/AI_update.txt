# AI科研船BUG修复
namespace = ethic_ai_science_patch
country_event = {
    id = ethic_ai_science_patch.1
    hide_window = yes
    is_triggered_only = yes

    trigger = {
        is_ai = yes
    }

   immediate = {
       every_owned_fleet = {
           limit = {
               OR = {
                   is_ship_size = science
                   is_ship_size = explorer_vessel
               }
               NOT = { exists = leader }
           }
           owner = {
               if = {
                   limit = {
                       any_owned_leader = {
                           leader_class = scientist
                           is_councilor = no
                           is_ruler = no
                       }
                   }
                   random_owned_leader = {
                       limit = {
                           leader_class = scientist
                           is_councilor = no
                           is_ruler = no
                       }
                       save_event_target_as = ai_scientist
                   }
               }
               else = {
                   create_leader = {
                       class = scientist
                       name = random
                       species = owner_main_species
                   }
                   last_created_leader = {
                       save_event_target_as = ai_scientist
                       if = {
                           limit = {
                               has_total_skill > 1
                               num_traits = 0
                           }
                           add_trait = {
                               trait = random_common
                               show_message = no
                           }
                       }
                   }
               }
           }
           set_leader = event_target:ai_scientist
       }
   }
}

# AI人口补强
namespace = ethic_ai_owned_pop_amount
country_event = {
    id = ethic_ai_owned_pop_amount.1
    hide_window = yes
    is_triggered_only = yes

    trigger = {
        is_ai = yes
    }

    immediate = {
        every_owned_planet = {
            if = {
                limit = {
                    count_owned_pop_amount = {
                        count < 10000
                    }
                }
                create_pop_group = {
                    size = 1000
                    species = owner_main_species
                }
            }
            if = {
                limit = {
                    count_owned_pop_amount = {
                        count >= 10000
                    }
                }
                create_pop_group = {
                    size = 2000
                    species = owner_main_species
                }
            }
        }
    }
}
