ap_become_the_crisis = {
	potential = {
		has_nemesis = yes
		NOT = {
		    has_ascension_perk = ap_become_the_crisis
		}
		NOT = {
			has_ascension_perk = ap_defender_of_the_galaxy
		}
	}

	possible = {
		custom_tooltip = {
			fail_text = "requires_ascension_perks_3"
			OR = {
				num_ascension_perks > 2
				has_origin = origin_galactic_nemesis
			}
		}
		custom_tooltip = {
			fail_text = "requires_independence"
			is_subject = no
		}
		custom_tooltip = {
			fail_text = "must_not_be_custodian"
			is_galactic_custodian = no
		}
		custom_tooltip = {
			fail_text = "must_not_be_emperor"
			is_galactic_emperor = no
		}
		custom_tooltip = {
			fail_text = "requires_not_ap_defender_of_the_galaxy"
			NOT = { has_ascension_perk = ap_defender_of_the_galaxy }
		}
		NOR = {
			has_ethic = ethic_xenophile
			has_ethic = ethic_fanatic_xenophile
			has_ethic = ethic_pacifist
			has_ethic = ethic_fanatic_pacifist
			has_civic = civic_machine_servitor
			has_tradition = tr_tt_liberation_adopt
		}
	}

	on_enabled = {
		hidden_effect = {
			set_ai_personality = became_the_crisis
		}
		activate_crisis_progression = yes
	}

	ai_weight = {
		factor = 7.5
		modifier = {
			factor = 0
			is_militarist = no
			is_xenophobe = no
			is_gestalt = no
			NOT = { has_origin = origin_galactic_nemesis }
		}
		modifier = {
			factor = 0
			OR = {
				has_valid_civic = civic_shared_burden
				has_valid_civic = civic_advanced_socialism
			}
		}
		modifier = {
			factor = 0
			has_valid_civic = civic_advanced_socialism
		}
		modifier = {
			factor = 10
			has_valid_civic = civic_nazbol
		}
		modifier = {
			factor = 10
			is_homicidal = yes
		}
		modifier = {
			factor = 10000
			has_origin = origin_galactic_nemesis
		}
		modifier = {
			factor = 10000
			is_militarist = yes
			is_xenophobe = yes
			has_ascension_perk = ap_conquer_the_unknown
		}
		modifier = {
			factor = 2
			has_tradition = tr_supremacy_finish
		}
		modifier = {
			factor = 0
			galaxy_percentage < 0.1
			NOR = {
				has_origin = origin_galactic_nemesis
				is_homicidal = yes
			}
		}
		modifier = {
			factor = 2
			galaxy_percentage > 0.1
			is_homicidal = yes
		}
		modifier = {
			factor = 2
			galaxy_percentage > 0.2
		}
		modifier = {
			factor = 0
			count_playable_country = {
				count > 1
				limit = {
					has_ascension_perk = ap_become_the_crisis
				}
			}
			NOT = { has_origin = origin_galactic_nemesis }
		}
	}
}
