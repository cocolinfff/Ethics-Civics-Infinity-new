tr_noble_adopt = {
	unlocks_agenda = agenda_tr_noble
    custom_tooltip = tr_noble_adopt_desc
	possible = {
		NOT = { has_ethic = ethic_fanatic_militarist }
	}
	on_enabled = {
		if = {
			limit = { is_at_war = no }
			add_modifier = { modifier = mod_tr_noble_adopt days = -1 }
		}
	}
	
	modifier = {
		planet_crime_mult = -0.25
	}
}
