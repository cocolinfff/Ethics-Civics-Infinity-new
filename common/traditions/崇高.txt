tr_noble_adopt = {
	possible = {
		NOT = { has_ethic = ethic_fanatic_militarist }
	}
	on_enabled = {
		if = {
			limit = { is_at_war = no }
			add_modifier = { modifier = mod_tr_noble_adopt days = -1 }
		}
	}
	
	modifier = {
		planet_crime_mult = -0.25
	}
}

tr_noble_finish = {

	}

		pop_ethic_pacifist_attraction_mult = 0.10
