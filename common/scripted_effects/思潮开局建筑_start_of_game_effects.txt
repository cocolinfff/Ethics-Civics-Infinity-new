civic_true_yuht_start_buildings = {
	optimize_memory
	owner = {
		save_event_target_as = civic_true_yuht_start
		capital_scope = {
			add_deposit = d_machine_minerals
			solar_system = {
				every_system_planet = {
					limit = {
						colonizable_planet = no
						is_star = no
					}
					clear_deposits = yes
				}
				every_system_planet = {
					limit = {
						is_asteroid = yes
					}
					remove_planet = yes
				}
				every_system_planet = {
					limit = {
						colonizable_planet = no
						is_star = no
						is_asteroid = no
						NOT = { is_planet_class = pc_gas_giant }
					}
					change_pc = pc_broken
				}
				every_system_planet = {
					limit = {
						colonizable_planet = no
						is_star = no
						is_planet_class = pc_gas_giant
					}
					change_pc = pc_molten
					add_modifier = { modifier = chthonian_planet }
					every_moon = {
						limit = {
							colonizable_planet = no
						}
						remove_planet = yes
					}
					while = {
						limit = {
							planet_size >= 20
						}
						change_planet_size = -6
					}
				}
			}
		}
	}
	solar_system = {
		every_system_colony = {
			limit = {
				exists = owner
				owner = { has_origin = origin_true_yuht }
			}
			clear_blockers = yes
			remove_all_buildings = yes
			remove_all_districts = yes
			add_modifier = {
				modifier = mineral_rich
				days = -1
			}
			add_deposit = d_dust_caverns
			add_deposit = d_bubbling_swamp
			add_deposit = d_crystalline_caverns
			add_modifier = { modifier = satellite_city days = -1 }
			add_modifier = { modifier = space_industry_cluster days = -1 }
			add_modifier = { modifier = space_industry_cluster_industrial days = -1 }
			add_modifier = { modifier = space_industry_cluster_research days = -1 }

			while = {
				count = 20
				add_district_and_planet_size_if_needed_effect = {
					district = district_mining
				}
			}
			if = {
				limit = { owner = { is_machine_empire = yes } }
				while = {
					count = 20
					add_district_and_planet_size_if_needed_effect = {
						district = district_nexus
					}
				}
			}
			if = {
				limit = { owner = { is_hive_empire = yes } }
				while = {
					count = 20
					add_district_and_planet_size_if_needed_effect = {
						district = district_hive
					}
				}
			}
			if = {
				limit = { owner = { is_gestalt = no } }
				while = {
					count = 20
					add_district_and_planet_size_if_needed_effect = {
						district = district_city
					}
				}
			}
			add_zone = {
				district = district_mining
				zone = zone_minerals
			}
			add_building = {
				district = district_mining
				zone = zone_minerals
				building = building_mine_generic
			}
			add_building = {
				district = district_mining
				zone = zone_minerals
				building = building_mineral_purification_hub
			}
			add_building = {
				district = district_mining
				zone = zone_minerals
				building = building_mining_districts_3
			}

			if = {
				limit = {
					owner = {
						is_gestalt = no
						NOT = { has_country_flag = eci_bug_gongjuren_flag }
					}
				}
				add_building = building_system_capital
			}

				add_building = building_ancient_palace
			}

			}
