
d_eci_United_States = {
    icon = d_broken_spire
    time = 1800

    is_for_colonizable = yes
    category = deposit_cat_blockers

    potential = { always = no }

    planet_modifier = {
        planet_max_districts_add = -1
        planet_jobs_trade_produces_mult = 0.5
        planet_stability_mult = -0.5
        pop_bonus_workforce_mult = -0.1
    }

    on_cleared = {
        hidden_effect = {
            planet.owner = {
                country_event = { id = ethic_advanced_start_independent.1 }
            }
            if = {
                limit = {
                    any_owned_pop_group = {
                        NOT = { has_trait = trait_hive_mind }
                        is_robot_pop_group = no
                    }
                }
                weighted_random_owned_pop_group = {
                    limit = {
                        NOT = { has_trait = trait_hive_mind }
                        is_robot_pop_group = no
                    }
                    root = {
                        while = { 
                            count = 2
                            create_pop_group = {
                                species = prev
                            }
                        }
                    }
                }
            }
            else = {
                weighted_random_owned_pop_group = {
                    root = {
                        while = {
                            count = 2
                            create_pop_group = {
                                species = prev
                            }
                        }
                    }
                }
            }
        }
        custom_tooltip = d_decrepit_dwellings_earth_eci_custom_tooltip
    }

    inline_script = deposits/ai_weight_pop_creating_blocker

    resources = {
        category = deposit_blockers
        cost = {
            influence = 2000
        }
    }

}
