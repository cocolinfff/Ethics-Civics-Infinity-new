jhh_armies_on_planet = {
    AND = {
        count_planet_army = {
            count <= $RANGE_MAX$
            limit = {
                exists = owner
                exists = ROOT.owner
                ROOT.owner = { has_overlord = PREV.owner }
            }
        }
        count_planet_army = {
            count >= $RANGE_MIN$
            limit = {
                exists = owner
                exists = ROOT.owner
                ROOT.owner = { has_overlord = PREV.owner }
            }
        }
    }
}
