wg_imperial_war = {
	casus_belli = cb_subjugation
	set_defender_wargoal = wg_imperial_war

	hide_if_no_cb = yes
	total_war = yes
	release_occupied_systems_on_status_quo = yes

	allowed_peace_offers = {
		status_quo
		surrender
		demand_surrender
	}

	surrender_acceptance = -100

	war_exhaustion = 0.75

	potential = {
		OR = {
			AND = {
				OR = {
					has_country_flag = government_in_power
					has_country_flag = opposition_rebel
				}
				from = {
					OR = {
						has_country_flag = government_in_power
						has_country_flag = opposition_rebel
					}
				}
			}
			AND = {
				has_country_flag = Broken_Red_Alliance
				from = {
					has_country_flag = Broken_Red_Alliance
				}
			}
		}
	}

	on_status_quo = {
		from = {
			while = {
				random_system = {
					add_claims = { who = root }
				}
			}
			if = {
				limit = { has_country_flag = government_in_power }
				remove_country_flag = government_in_power
			}
		}
		root = {
			while = {
				random_system = {
					add_claims = { who = from }
				}
			}
			if = {
				limit = { has_country_flag = government_in_power }
				remove_country_flag = government_in_power
			}
		}
	}

	ai_weight = {
		weight = 30
		modifier = {
			factor = 0
			NOT = {
				has_country_flag = Broken_Red_Alliance
				from = {
					has_country_flag = Broken_Red_Alliance
				}
			}
		}
	}
}

wg_capture_the_market = {
	casus_belli = cb_the_benefits_spread

	hide_if_no_cb = no
	release_occupied_systems_on_status_quo = yes

	allowed_peace_offers = {
		status_quo
		surrender
		demand_surrender
	}

	surrender_acceptance = -100

	war_exhaustion = 0.75

	potential = {
		OR = {
			has_valid_civic = civic_merchant_guilds
			is_eci_megacorp = yes
		}
		any_galaxy_planet = {
			has_branch_office = yes
			branch_office_owner = {
				is_same_value = from
			}
		}
		NOT = { is_in_federation_with = from }
		NAND = {
			is_galactic_emperor = yes
			from = { is_galactic_community_member = yes }
		}
	}
	possible = {
		custom_tooltip = {
			fail_text = "cannot_use_against_emperor"
			from = { is_galactic_emperor = no }
		}
		custom_tooltip = {
			fail_text = pax_galactica_active
			NAND = {
				is_active_resolution = resolution_emperor_pax_galactica
				is_galactic_community_member = yes
				from = { is_galactic_community_member = yes }
			}
		}
	}
	on_status_quo = {
		from = {
			add_threat = {
				who = root
				amount = 0.05
			}
		}
	}
	on_accept = {
		custom_tooltip = wg_capture_the_market_accept
		hidden_effect = {
			every_galaxy_planet = {
				limit = {
					has_branch_office = yes
					branch_office_owner = { is_same_value = from }
				}
				establish_branch_office = root
			}
		}
		from = {
			add_threat = {
				who = root
				amount = 0.10
			}
		}
	}

	ai_weight = {
		weight = 100
	}
}

wg_independence = {
	casus_belli = cb_subject

	hide_if_no_cb = yes
	set_defender_wargoal = wg_imperial_war

	potential = {
		always = yes
	}
	possible = {
		custom_tooltip = {
			fail_text = cannot_use_against_emperor
			OR = {
				from = { is_galactic_emperor = no }
				is_galactic_community_member = no
			}
		}
		custom_tooltip = {
			fail_text = pax_galactica_active
			NAND = {
				is_active_resolution = resolution_emperor_pax_galactica
				is_galactic_community_member = yes
				from = { is_galactic_community_member = yes }
			}
		}
	}

	allowed_peace_offers = {
		status_quo
		surrender
		demand_surrender
	}

	on_status_quo = {
		custom_tooltip = wg_independence_status_quo
		hidden_effect = {
			fromfrom = {
				every_war_participant = {
					limit = {
						fromfrom = {
							is_war_participant = {
								who = prev
								side = Root
							}
						}
						has_overlord = Root.from
					}
					set_subject_of = {
						who = none
					}
				}
			}
		}
	}

	on_accept = {
		custom_tooltip = wg_independence_accept
		hidden_effect = {
			fromfrom = {
				every_war_participant = {
					limit = {
						fromfrom = {
							is_war_participant = {
								who = prev
								side = Root
							}
						}
						has_overlord = Root.from
					}
					set_subject_of = {
						who = none
					}
					every_system_within_border = {
						limit = {
							Root.from = { has_claim = prev }
						}
						remove_claims = { who = Root.from }
					}
				}
			}
		}
	}

	on_wargoal_set = {
		every_system_within_border = {
			limit = {
				NOT = { from = { has_claim = prev } }
			}
			add_claims = { who = from }
		}
	}

	ai_weight = {
		weight = 100
	}

	should_ignore_federations = yes
}
