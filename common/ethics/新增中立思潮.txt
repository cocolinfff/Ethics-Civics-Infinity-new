#调和主义
ethic_conciliationsm = {
	cost = 1
	category = "col"
	category_value = 2

	tags = {
		ETHIC_AUTHORITARIAN_UNITY
		ETHIC_EGALITARIAN_UNITY
		ETHIC_ALLOWS_SHARED_BURDEN
		ETHIC_ALLOWS_STRATIFIED_SOCIETY
	}

	country_modifier = {
		pop_ethic_conciliationsm_attraction_mult = 1
		show_only_custom_tooltip = no
		add_attunement_the_cradle_of_souls = @ethic_attunement1
	}

	leader_background_job_weight = {
		core_party_members = 100
		noble = 100
		sovereign_representative = 100
		bureaucrat = 50
		clerk = 25
		miner = 25
		crystal_miner = 25
		farmer = 25
		preacher = 25
		scrap_miner = 25
	}

	random_weight = {
		base = 100
	}

	pop_attraction_tag = {
		desc = POP_ATTRACTION_CONCILIATIONSM_POS
		trigger = {
			is_conciliationsm = yes
		}
	}

	pop_attraction_tag = {
		desc = POP_ATTRACTION_FACTION_POS
		trigger = {
			is_conciliationsm = no
			has_faction = silent
		}
	}

	pop_attraction_tag = {
		desc = POP_ATTRACTION_OVERLORD_POS
		trigger = {
			is_subject = yes
			has_loyalty > 0
			overlord = { is_conciliationsm = yes }
		}
	}

	# Check for any leader on the Council with Faction's Ethic.
	pop_attraction_tag = {
		desc = POP_ATTRACTION_REPRESENTED_ON_COUNCIL
		trigger = {
			owner = {
				any_owned_leader = {
					OR = {
						is_ruler = yes
						is_councilor = yes
					}
					has_ethic = ethic_conciliationsm
				}
			}
		}
	}

	pop_attraction_tag = {
		desc = POP_ATTRACTION_GOVERNOR_POS
		trigger = {
			any_owned_planet = {
				exists = leader
				leader = { has_ethic = ethic_conciliationsm }
			}
		}
	}

	country_attraction = {
		base = 2

		modifier = {
			factor = 2
			is_country_type = default
			is_conciliationsm = no
			has_faction = silent
		}

		modifier = {
			factor = 5
			has_ethic = ethic_conciliationsm
		}

		modifier = {
			mult = 2
			exists = overlord
			overlord = {
				has_ethic = ethic_conciliationsm
			}
		}

		# Check for any leader on the Council with Faction's Ethic.
		modifier = {
			factor = value:count_councilors_with_ethic|ETHIC|conciliationsm|
		}
	}
}
