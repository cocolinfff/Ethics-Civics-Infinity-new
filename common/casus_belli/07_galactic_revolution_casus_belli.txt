cb_galactic_revolution = {
	show_in_diplomacy = yes
	show_notification = no
	potential = {
		is_country_type = default
		OR = {
			is_true_scientific_socialism = yes
			has_civic = civic_machine_liberator
		}
		exists = from
		NOR = { is_in_federation_with = from }
	}
	is_valid = {
		has_policy_flag = continuous_revolution
		from = { is_true_scientific_socialism = no }
		NOT = { is_overlord_to = FROM }
		OR = {
			is_subject = no
			overlord = { NOT = { is_same_value = from } }
		}
	}
	proxy_war_resources = {
		category = proxy_war
		cost = {
			influence = 1000
		}
	}
	on_proxy_war_start = {
		pay_proxy_war_additional_resources = { VALUE = -25 }
	}
}
