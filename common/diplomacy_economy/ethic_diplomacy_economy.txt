once = {
	# this is applied only once per country
	federation = {
		trigger = {
			has_federation = yes

				has_valid_civic = civic_peacekeepers
			}
		}
		resources = {
			category = federation
			upkeep = {
				influence = 1
			}
		}
	}
}
