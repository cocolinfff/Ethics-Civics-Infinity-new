# Defense Armies
defense_army = {
	defensive = yes
	is_pop_spawned = yes
	health = 1.25
	damage = 1.50
	morale = 1.25
	collateral_damage = 0.0
	war_exhaustion = 0.0 # No WE from defense armies
	icon = GFX_army_type_defensive

	resources = {
		category = armies
		produces = {
			trigger = {
				exists = owner
				owner = { has_active_tradition = tr_unyielding_resistance_is_frugal }
			}
			unity = 0.5
		}
		produces = {
			trigger = {
				has_councilor = { COUNCILOR = "councilor_army_production" }
			}
			unity = 0.1
			mult = value:councilor_level|TYPE|councilor_army_production|
		}
		produces = {
			trigger = {
				has_councilor = { COUNCILOR = "councilor_army_production_corporate" }
			}
			unity = 0.1
			mult = value:councilor_level|TYPE|councilor_army_production_corporate|
		}
		produces = {
			trigger = {
				NOT = { from = { has_trait = trait_lithoid } }
				exists = owner
				owner = { has_valid_civic = civic_martial_brotherhood }
			}
			unity = 0.5
			energy = 2
			minerals = 2
			food = 2
		}
		produces = {
			trigger = {
				from = { has_trait = trait_lithoid }
				exists = owner
				owner = { has_valid_civic = civic_martial_brotherhood }
			}
			unity = 0.5
			energy = 2
			minerals = 4
		}
	}

	potential = {
		from = {
			NOR = {
				has_trait = "trait_mechanical"
				has_trait = "trait_machine_unit"
				is_sapient = no
			}
		}
		owner = {
			is_primitive = no
			OR = {
				NOT = { has_authority = auth_machine_intelligence }
				has_valid_civic = civic_machine_assimilator
			}
		}
	}
}
