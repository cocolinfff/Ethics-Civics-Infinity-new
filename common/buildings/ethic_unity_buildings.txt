		building_hive_node

		building_planet_network_node

		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = 2
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			always = yes
		}
		modifier = {
			job_bureaucrat_add = 2
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner 
			is_eci_megacorp = yes
		}
		modifier = {
			job_bureaucrat_add = -2
			job_manager_add = 2
		}
	}

	triggered_desc = {
		trigger = {
			exists = owner
			NOT = { is_eci_megacorp = yes }
		}
		text = job_bureaucrat_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			is_eci_megacorp = yes
		}
		text = job_manager_effect_desc
	}

	upgrades = {
		"building_bureaucratic_2"
	}

	prerequisites = {
		"tech_planetary_government"
	}

	show_tech_unlock_if = {
		is_regular_empire = yes
	}
}
