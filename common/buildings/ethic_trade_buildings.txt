building_commercial_zone = {
	base_buildtime = @b1_time
	category = trade
	potential = {
		exists = owner
		owner = {
			is_regular_empire = yes
		}
		NOR = {
			has_modifier = slave_colony
			has_building = building_low_tech_capital
			has_planet_flag = planetary_station
		}
	}
	allow = {
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_valid_civic = civic_eternal_jihad
				is_duelist_country = no
			}
		}
		modifier = {
			job_professional_star_fans_add = 2
			job_entertainer_add = 1
			pop_ethic_capitalism_attraction_mult = 0.05
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_valid_civic = civic_eternal_jihad
				is_duelist_country = yes
			}
		}
		modifier = {
			job_professional_star_fans_add = 2
			job_duelist_add = 1
			pop_ethic_capitalism_attraction_mult = 0.05
			pop_ethic_militarist_attraction_mult = 0.05
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				NOT = { has_valid_civic = civic_eternal_jihad}
			}
		}
		modifier = {
			job_shopkeeper_add = 3
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				OR = {
					has_valid_civic = civic_consumerism
				}
			}
		}
		modifier = {
			job_artisan_add = 1
		}
	}
	destroy_trigger = {
		exists = owner
		OR = {
			owner = {
				is_regular_empire = no
			}
			has_modifier = slave_colony
		}
	}
	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = @b1_upkeep
		}
	}
	prerequisites = {
		"tech_interplanetary_commerce"
	}
	show_tech_unlock_if = {
		is_regular_empire = yes
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				has_valid_civic = civic_eternal_jihad
			}
		}
		text = job_professional_star_fans_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				NOR = {
					has_ethic = ethic_fanatic_socialism
					has_valid_civic = civic_eternal_jihad
				}
			}
		}
		text = job_shopkeeper_effect_desc
	}
	upgrades = {
		"building_commercial_megaplex"
	}
	ai_resource_production = {
		energy = 2
		trigger = {
			always = yes
		}
	}
	ai_resource_production = {
		energy = 3
		trigger = {
			OR = {
				num_unemployed >= 10
				num_assigned_jobs = {
					job = servant
					value >= 10
				}
			}
		}
	}
}

#银河证券交易所
building_galactic_stock_exchange = {
	base_buildtime = @b2_time
	base_cap_amount = 1
	category = trade
	potential = {
		NOR = {
			has_modifier = resort_colony
			has_modifier = slave_colony
			has_planet_flag = planetary_station
		}
	}
	allow = {
		has_major_upgraded_capital = yes
		if = {
			limit = {
				exists = orbital_defence
			}
			orbital_defence = {
				exists = starbase
				starbase = {
					NOR = {
						has_starbase_building = ring_galactic_stock_exchange
						is_starbase_building_building = ring_galactic_stock_exchange
					}
				}
			}
		}
	}
	destroy_trigger = {
		exists = owner
		OR = {
			owner = {
				is_regular_empire = no
			}
			has_modifier = slave_colony
			has_modifier = resort_colony
		}
	}
	planet_modifier = {
		job_merchant_add = 2
		trade_value_mult = 0.2
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				OR = {
					has_valid_civic = civic_merchant_guilds
					has_valid_civic = civic_state_monopoly
				}
			}
		}
		modifier = {
			job_mogul_add = 1
			trade_value_mult = 0.05
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				OR = {
					has_valid_civic = civic_anarchism
				}
			}
		}
		modifier = {
			job_merchant_add = 2
		}
	}
	resources = {
		category = planet_buildings
		cost = {
			minerals = @b2_minerals
			rare_crystals = @b2_rare_cost
		}
		upkeep = {
			energy = @b2_upkeep
			rare_crystals = @b2_rare_upkeep
		}
	}
	prerequisites = {
		"tech_galactic_markets"
	}
	triggered_desc = {
		trigger = {
			exists = owner
		}
		text = job_merchant_effect_desc
	}
	ai_resource_production = {
		energy = 3
		trigger = {
			always = yes
		}
	}
	ai_resource_production = {
		energy = 5
		trigger = {
			solar_system = {
				has_system_trade_value >= 50
			}
		}
	}
	ai_resource_production = {
		energy = 3
		trigger = {
			solar_system = {
				has_system_trade_value >= 80
			}
		}
	}
}
