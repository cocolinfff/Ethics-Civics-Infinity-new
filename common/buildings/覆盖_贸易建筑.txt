		modifier = {
			factor = 10
			owner = { has_policy_flag = economic_stance_market }
		}
	}

#银河证券交易所
building_galactic_stock_exchange = {
	base_buildtime = @b2_time
	base_cap_amount = 1
	category = trade
	potential = {
		NOR = {
			has_modifier = resort_colony
			has_modifier = slave_colony
			has_planet_flag = planetary_station
		}
	}
	allow = {
		has_major_upgraded_capital = yes
		if = {
			limit = {
				exists = orbital_defence
			}
			orbital_defence = {
				exists = starbase
				starbase = {
					NOR = {
						has_starbase_building = ring_galactic_stock_exchange
						is_starbase_building_building = ring_galactic_stock_exchange
					}
				}
			}
		}
	}
	destroy_trigger = {
		exists = owner
		OR = {
			owner = {
				is_regular_empire = no
			}
			has_modifier = slave_colony
			has_modifier = resort_colony
		}
	}
	planet_modifier = {
		job_mogul_add = 2
		trade_value_mult = 0.2
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_valid_civic = civic_merchant_guilds
			}
		}
		modifier = {
			job_mogul_add = 1
			trade_value_mult = 0.05
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				OR = {
					has_valid_civic = civic_anarchism
					has_valid_civic = civic_beacon_of_liberty
				}
			}
		}
		modifier = {
			job_mogul_add = -2
			job_trader_add = 5

		}
	}

		category = planet_buildings

		}
	}
	prerequisites = {
		"tech_galactic_markets"
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				NOR = {
					has_valid_civic = civic_anarchism
					has_valid_civic = civic_beacon_of_liberty
				}
			}
		}
		text = job_mogul_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				OR = {
					has_valid_civic = civic_anarchism
					has_valid_civic = civic_beacon_of_liberty
				}
			}
		}
		text = job_trader_effect_desc
	}
	ai_resource_production = {
		energy = 20

			always = yes
		}
