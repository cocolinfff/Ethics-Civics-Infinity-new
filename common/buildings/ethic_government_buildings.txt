					}

							has_building = building_precinct_house
							owner = {
								has_technology = tech_colonial_centralization
							}
						}
						has_building = building_hall_judgment
						num_buildings = {

							value = 3
						}
					}
				}
			}
		}

	destroy_trigger = {
		exists = owner

			NOT = {
				owner = {

				}
			}
			AND = {
				owner = {

					has_building = building_hall_judgment
				}
			}

	triggered_planet_modifier = {

		}
